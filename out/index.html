Getting started

Installation

npm i react-mui-scheduler

Usage

import Scheduler, { Mode, Option, StartWeek, TransitionMode } from "react-mui-scheduler";
// Import i18Next configuration
import "react-mui-scheduler/dist/locales/i18n";

const events: Event[] = [
  {
    id: "event-1",
    label: "Medical consultation",
    groupLabel: "Dr Shaun Murphy",
    user: "Dr Shaun Murphy",
    color: "#f28f6a",
    startDate: "04:00 AM",
    endDate: "06:00 AM",
    date: "2023-01-01",
    createdAt: new Date(),
    createdBy: "Kristina Mayer",
  },
];

const options: Option = {
  transitionMode: TransitionMode.ZOOM,
  startWeekOn: StartWeek.SUN,
  defaultMode: Mode.MONTH,
  minWidth: 540,
  maxWidth: 540,
  minHeight: 540,
  maxHeight: 540,
  reverseTimelineOrder: false,
};

<Scheduler
  locale="en"
  events={ events }
  options={ options }
/>
